*** Calculando valores com Left join + Sum + Count e Group By
====================================
SELECT 
    c.nome_cargo,
    SUM(c.valor_cargo) AS Valor_Pago,
    COUNT(f.cod_cargo_funcionario) AS Total_Funcionarios
FROM
    itau.cargo AS c
        LEFT JOIN
    itau.funcionario AS f ON (f.cod_cargo_funcionario = c.cod_cargo)
    group by c.cod_cargo;

    SUM(quantidade_produto * valor_unitario) AS Total_Produtos,
